163 template <class FromComponentType, class FromOutputId, class ToComponentType, class ToInputId>
164 bool DspCircuit::ConnectOutToIn(FromComponentType& fromComponent,
165  FromOutputId const& fromOutput,
166  ToComponentType& toComponent,
167  ToInputId const& toInput)
168 {
169  int fromComponentIndex;
170  int toComponentIndex;
171 
172  // only interconnect components that have been added to this system
173  if (!_FindComponent(fromComponent, fromComponentIndex) || !_FindComponent(toComponent, toComponentIndex))
174  {
175  return false;
176  }
177 
178  PauseAutoTick();
179 
180  bool result = _components[toComponentIndex]->ConnectInput(_components[fromComponentIndex], fromOutput, toInput);
181 
182  ResumeAutoTick();
183 
184  return result;
185 }

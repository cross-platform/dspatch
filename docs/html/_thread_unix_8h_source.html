70  virtual void Start( Priority priority = NormalPriority )
71  {
72  pthread_create( &_thread, nullptr, _ThreadFunc, this );
73  _threadAttatched = true;
74 
75  _SetPriority( _thread, priority );
76  }
77 
78  virtual void Stop()
79  {
80  if ( _threadAttatched )
81  {
82  pthread_detach( _thread );
83  _threadAttatched = false;
84  }
85  }
86 
87 protected:

89 
90 private:
91  static void* _ThreadFunc( void* pv )
92  {
93  ( reinterpret_cast<Thread*>( pv ) )->Run_();
94  return nullptr;
95  }

97  static void _SetPriority( pthread_t threadID, Priority priority )
98  {
99  int policy;
100  struct sched_param param;
101 
102  pthread_getschedparam( threadID, &policy, &param );
103 
104  policy = SCHED_FIFO;
105  param.sched_priority = ( ( priority - IdlePriority ) * ( 99 - 1 ) / TimeCriticalPriority ) + 1;
106 
107  pthread_setschedparam( threadID, policy, &param );
108  }
